/**
 * Checks if the person is logged
 * @returns {string} true, false or undefined
 */

function get_logged(){
    const fs = require('fs'); 
    return fs.readFileSync('logged', 'utf-8'); 
}

function set_logged(logged){
    const fs = require('fs'); 
    fs.writeFileSync('logged', logged); 
}

module.exports = {get_logged, set_logged};
