/**
 * Handler function to call the python script
 */
function handleTheacherSched() {
    let code = document.getElementById("docentCode").value;

    // get the correct date
    let year = new Date().getFullYear();
    let month = new Date().getMonth();
    if (month <= 7) year -= 1;

    cleanMessageError();
    addLoadAnimation();
    pyCall("search_teacher_schedule", "final_handleTeacherSched", [code, year]);

}

/**
 * Function to be called after python script get's the professor's schedule.
 * @param data - python output dictionary
 */
function final_handleTeacherSched(data) {
    removeLoadAnimation();
    console.log(data);
    if (data['error'] === 'true') errorMessage();
    else {
        console.log(data);
        document.getElementById('professor-name').innerHTML = data['name'];
        document.getElementById('table-wrapper').innerHTML = data['table'];
        treatTable();

    }
}

function errorMessage() {
    console.log('ERROR');
    document.getElementById('error').innerHTML = "<span>Docente inexistente ou docente não possui horário</span>";
}

function cleanMessageError() {
    document.getElementById('error').innerHTML = "";
}

/*
 *
 * The table that come from sigarra needs to be treated to be well
 * formated to the module build it
 */
function treatTable() {
    document.getElementsByTagName('table')[0].setAttribute("class", "table table-condensed table-hover table-striped  table-sm time");

    // change the hrefs for ones that works
    let aTags = document.getElementsByTagName('table')[0].getElementsByTagName('a');
    for (let i = 0; i < aTags.length; i++) {
        let newHref = "https://sigarra.up.pt/feup/pt/" + aTags[i].getAttribute('href');
        aTags[i].setAttribute('href', newHref);

    }
}
